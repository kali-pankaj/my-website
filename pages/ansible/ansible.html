Ansible complete guide

Introduction to Ansible

Ansible is an open-source automation tool used for configuration management, application deployment, and infrastructure automation.

Ansible Setup & Commands

  • sudo apt update && sudo apt install ansible -y – Install Ansible.
  • ansible --version – Check installed version.
  • ansible all -m ping -i inventory.ini – Test connectivity.
  • ansible-playbook playbook.yml -i inventory.ini – Run playbook.
  • ansible-galaxy install role_name – Install role from Galaxy.

Useful Full Commands

  • ansible-playbook --syntax-check playbook.yml
  • ansible localhost -m ping
  • ansible all -m ping
  • ansible 192.168.1.10 -m ping
  • ansible all -m copy -a "src=/path/file dest=/tmp/file"
  • ansible all -m service -a "name=nginx state=reloaded"
  • ansible all -m shell -a "/tmp/script.sh"
  • ansible all -m command -a "free -h"
  • ansible all -m apt -a "name=vim state=present" -b

App Installation

File: install_apache.yml


      ---
      - name: Install and Start Apache
        hosts: localhost
        become: yes

        tasks:
          - name: Install Apache
            apt:
              name: apache2
              state: present

          - name: Start Apache
            service:
              name: apache2
              state: started
              enabled: yes

Run: ansible-playbook install_apache.yml

Copy File to Remote Server

File: copy_file_remote.yml

---
      - name: Copy file to remote server
        hosts: all
        become: yes

        tasks:
          - name: Copy file
            copy:
              src: /home/pankaj/Desktop/file.txt
              dest: /tmp/file.txt
              owner: pankaj
              group: pankaj
              mode: '0644'

Run: ansible-playbook copy_file_remote.yml

Create Directory, File, Set Permission & Remove (absent)

file: create_directory.yml

---
      - name: Manage directory and file
        hosts: localhost
        become: yes

        tasks:
          - name: Create directory
            file:
              path: /home/pankaj/ansible_test_dir
              state: directory
              owner: pankaj
              group: pankaj
              mode: '0755'

          - name: Create empty file
            file:
              path: /home/pankaj/ansible_test_dir/test.txt
              state: touch
              owner: pankaj
              group: pankaj
              mode: '0644'

          - name: Remove directory
            file:
              path: /home/pankaj/ansible_test_dir
              state: absent

Run: ansible-playbook create_directory.yml

User Management (Add User & Group)

file: user_management.yml

---
      - name: Manage users and groups
        hosts: all
        become: yes

        tasks:
          - name: Create group
            group:
              name: mygroup
              state: present

          - name: Create user
            user:
              name: myuser
              group: mygroup
              comment: "My Ansible User Account"
              shell: /bin/bash
              home: /home/myuser
              state: present

Run: ansible-playbook user_management.yml

User Password Management

file: set_user_password.yml

---
      - name: Set user password
        hosts: all
        become: yes

        tasks:
          - name: Update user password
            user:
              name: myuser
              update_password: always
              password: "{{ 'mypassword' | password_hash('sha512') }}"

Run: ansible-playbook set_user_password.yml

Variables in Ansible

File: ansible_variables.yml

---
    - name: Example Playbook with Variables
      hosts: all
      become: yes

      vars:
        packages:
          - nginx
          - mysql-server
          - vim

        web_service: nginx

      tasks:
        - name: Install Packages
          apt:
            name: "{{ packages }}"
            state: present
          tags:
            - install

        - name: Start Web Service
          service:
            name: "{{ web_service }}"
            state: started
            enabled: yes
          tags:
            - start

Run Command: ansible-playbook ansible_variables.yml

Handlers in Ansible

File: firewall.yml

---
    - name: Firewall Changes
      hosts: all
      become: yes

      tasks:
        - name: Enable HTTP Service in Firewalld
          firewalld:
            port: 80/tcp
            permanent: true
            state: enabled
          notify:
            - Reload firewalld

      handlers:
        - name: Reload firewalld
          service:
            name: firewalld
            state: reloaded

Run Command: ansible-playbook firewall.yml

Condition in Ansible

File: condition.yml


      ---
      - name: Install Web Server Based on OS
        hosts: all
        become: true

        tasks:
          - name: Install httpd on RedHat
            yum:
              name: httpd
              state: present
            when: ansible_os_family == "RedHat"

          - name: Install apache2 on Debian/Ubuntu
            apt:
              name: apache2
              state: present
            when: ansible_os_family == "Debian"

Run Command: ansible-playbook conditation.yml

User Create Using Loops

      ---
      - name: Create multiple users using loop
        hosts: all
        become: yes

        tasks:
          - name: Create users
            user:
              name: "{{ item }}"
              comment: "User created by Ansible"
              shell: /bin/bash
              state: present
            loop:
              - raju
              - ram
              - sam
              - lucky
              - pankaj

Run Command: ansible-playbook create_user.yml

Package Install Using Loop

File: package_install.yml


      ---
      - name: Install packages using loop
        hosts: all
        become: yes

        vars:
          packages:
            - nginx
            - httpd
            - firewalld
            - vim

        tasks:
          - name: Install packages on Debian
            apt:
              name: "{{ item }}"
              state: present
            loop: "{{ packages }}"
            when: ansible_os_family == "Debian"
            tags:
              - install

          - name: Install packages on RedHat
            yum:
              name: "{{ item }}"
              state: present
            loop: "{{ packages }}"
            when: ansible_os_family == "RedHat"
            tags:
              - install

Run Command: ansible-playbook package_install.yml
